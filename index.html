Color Convert

RGB --> HSL / HSV(HSB)

HSL --> RGB

HSV(HSB) --> RGB

使用方式GitHub

//使用方式：
//第一步，引入js文件

<script type="text/javascript" src="colorconvert.js"></script>

//第二步，调用$C(),或者CC()传入一个数组与转换规则

<script type="text/javascript">
    var rgb = [ 100, 23, 67 ];
    var res = $C( rgb, "rgb2hsl" );//支持rgb2hsl、rgb2hsv、hsv2rgb、hsl2rgb
    alert( res );
</script>
